    module Prism::Action

    Overview

    A callable HTTP action with Callbacks module included.

    NOTE (From API) The response #status_code and #headers must be configured before writing the response body. Once response output is written, changing the status and #headers properties has no effect.

    struct MyAction
  include Prism::Action

  def call
    text("ok")
  end

  after do
    p "MyAction: ok"
  end
end

MyAction.call(env)
# => "ok"

    macro halt!(status, response = nil) #

    Set HTTP status, close the response and stop the execution. Optionally specify response, otherwise print a default HTTP response for this status.

    def call
  halt!(403) # Will print "Unauthorized" into the response body
  text("ok") # This line will not be called
end

def call
  halt!(500, "Something's wrong!") # Will print "Something's wrong!" into the response body
end

def call
  halt!(409, {error: "Oops"}) # Will print "Oops" and set content type to JSON
end

def call
  halt!(403, PaymentError) # Will call #to_json on PaymentError
end
